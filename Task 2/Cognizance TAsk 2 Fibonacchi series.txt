This program prints Nth fibonacci number.
Function Main
    Declare Integer n   
    Output "Enter the value of n : "
    Input n
    Call Fibonacci(n)
End
Function Fibonacci (Integer n)
    Declare Integer a, b, c, i  
    Assign a = 0
    Assign b = 1
    If n<0
        Output "Invalid number."
    False:
        If n==0
            Output a
        False:
            If n==1
                Output b
            False:
                For i = 1 to n-1
                    Assign c = a+b
                    Assign a = b
                    Assign b = c
                End
                Output b
            End
        End
    End
End
